CMAKE_MINIMUM_REQUIRED(VERSION 3.1.3 FATAL_ERROR)

PROJECT(geodesic_grid_dvr)

SET(SRC_DIR ./src)
SET(HEADER )
SET(SOURCE ${SRC_DIR}/main.cpp)

INCLUDE_DIRECTORIES(
${GLEW_INC_DIR}
${FREEGLUT_INC_DIR}
${FREETYPE_INC_DIR}
)

LINK_DIRECTORIES(
${GLEW_LIB_DIR}
${FREEGLUT_LIB_DIR}
${FREETYPE_LIB_DIR}
)

ADD_EXECUTABLE(${PROJECT_NAME} ${HEADER} ${SOURCE})

TARGET_LINK_LIBRARIES(${PROJECT_NAME} glew32 freeglut)
